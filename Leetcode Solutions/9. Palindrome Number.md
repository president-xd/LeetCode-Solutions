## Solution
### Python = 
```python
class Solution(object):
    def isPalindrome(self, x):
        """
        :type x: int
        :rtype: bool
        """
        simple = str(x)
        reverse = ""

        for i in range(len(simple) - 1, -1, -1):
            reverse += simple[i]

        return reverse == simple

```

class Solution {
public:
    bool isPalindrome(int x) {
        string simple = to_string(x), reverse = "";
        for (int i = simple.length() - 1; i >= 0; i--){
            reverse += simple[i];
        }

        if (reverse == simple){
            return true;
        }
        else{
            return false;
        }
    }
};
